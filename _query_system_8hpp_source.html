153 template<typename T>
154 std::conditional_t<hasReadAccess<T>(), const T&, T&> getComponent(Entity entity)
155 {
156 const ComponentType typeIndex = getUniqueComponentTypeID<T>();
157 const auto it = m_componentArrays.find(typeIndex);
158
159 if (it == m_componentArrays.end())
160 THROW_EXCEPTION(InternalError, "Component array not found");
161
162 auto componentArray = std::static_pointer_cast<ComponentArray<T>>(it->second);
163
164 if (!componentArray)
165 THROW_EXCEPTION(InternalError, "Failed to cast component array");
166
167 if (!componentArray->hasComponent(entity))
168 THROW_EXCEPTION(InternalError, "Entity doesn't have requested component");
169 return componentArray->get(entity);
170 }
