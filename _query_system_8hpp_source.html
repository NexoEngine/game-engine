150 template<typename T>
151 std::conditional_t<hasReadAccess<T>(), const T&, T&> getComponent(Entity entity)
152 {
153 const ComponentType typeIndex = getUniqueComponentTypeID<T>();
154 const auto it = m_componentArrays.find(typeIndex);
155
156 if (it == m_componentArrays.end())
157 THROW_EXCEPTION(InternalError, "Component array not found");
158
159 auto componentArray = std::static_pointer_cast<ComponentArray<T>>(it->second);
160
161 if (!componentArray)
162 THROW_EXCEPTION(InternalError, "Failed to cast component array");
163
164 if (!componentArray->hasComponent(entity))
165 THROW_EXCEPTION(InternalError, "Entity doesn't have requested component");
166 return componentArray->get(entity);
167 }
