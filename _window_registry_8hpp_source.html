92 template<typename T>
93 requires std::derived_from<T, IDocumentWindow>
94 void registerWindow(std::shared_ptr<T> window)
95 {
96 auto &windowsOfType = m_windows[typeid(T)];
97 if (std::ranges::any_of(windowsOfType, [&](const auto &existingWindow) {
98 return existingWindow->getWindowName() == window->getWindowName();
99 }))
100 {
101 THROW_EXCEPTION(WindowAlreadyRegistered, typeid(T), window->getWindowName());
102 }
103 windowsOfType.push_back(window);
104 }

116 template<typename T>
117 requires std::derived_from<T, IDocumentWindow>
118 void unregisterWindow(const std::string &windowName)
119 {
120 const auto it = m_windows.find(typeid(T));
121 if (it == m_windows.end()) {
122 LOG(NEXO_WARN, "Window of type {} not found", typeid(T).name());
123 return;
124 }
125
126 auto &windowsOfType = it->second;
127 auto found = std::ranges::find_if(windowsOfType, [&windowName](const auto &w) {
128 return w->getWindowName() == windowName;
129 });
130
131 if (found == windowsOfType.end()) {
132 LOG(NEXO_WARN, "Window of type {} with name {} not found", typeid(T).name(), windowName);
133 return;
134 }
135
136 windowsOfType.erase(found);
137 }
