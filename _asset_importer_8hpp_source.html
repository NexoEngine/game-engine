250 template<typename AssetType> requires std::derived_from<AssetType, IAsset>
251 void AssetImporter::registerImporter(AssetImporterBase *importer, const int priority)
252 {
253 const auto typeIdx = std::type_index(typeid(AssetType));
254
255 if (!m_importers.contains(typeIdx)) {
256 m_importers[typeIdx] = {};
257 }
258 if (!m_importersDetails.contains(typeIdx)) {
259 m_importersDetails[typeIdx] = {};
260 }
261
262 auto& importersVec = m_importers[typeIdx];
263 auto& importersDetailsVec = m_importersDetails[typeIdx];
264
265 size_t i = 0;
266 for (; i < importersVec.size() && priority <= importersDetailsVec[i].priority ; ++i);
267 importersVec.insert(importersVec.begin() + static_cast<long>(i), importer);
268 importersDetailsVec.insert(importersDetailsVec.begin() + static_cast<long>(i), {priority});
269 }

272 AssetRef<AssetType> AssetImporter::importAsset(const AssetLocation& location,
273 const ImporterInputVariant& inputVariant)
274 {
275 auto importers = getImportersForType<AssetType>();
276 if (importers.empty()) {
277 LOG(NEXO_ERROR, "No importers registered for asset type {}", typeid(AssetType).name());
278 return GenericAssetRef::null().as<AssetType>();
279 }
280 return importAssetTryImporters(location, inputVariant, importers).template as<AssetType>();
281 }
