246 template<typename AssetType> requires std::derived_from<AssetType, IAsset>
247 void AssetImporter::registerImporter(AssetImporterBase *importer, const int priority)
248 {
249 const auto typeIdx = std::type_index(typeid(AssetType));
250
251 if (!m_importers.contains(typeIdx)) {
252 m_importers[typeIdx] = {};
253 }
254 if (!m_importersDetails.contains(typeIdx)) {
255 m_importersDetails[typeIdx] = {};
256 }
257
258 auto& importersVec = m_importers[typeIdx];
259 auto& importersDetailsVec = m_importersDetails[typeIdx];
260
261 size_t i = 0;
262 for (; i < importersVec.size() && priority <= importersDetailsVec[i].priority ; ++i);
263 importersVec.insert(importersVec.begin() + static_cast<long>(i), importer);
264 importersDetailsVec.insert(importersDetailsVec.begin() + static_cast<long>(i), {priority});
265 }

267 template<typename AssetType> requires std::derived_from<AssetType, IAsset>
268 AssetRef<AssetType> AssetImporter::importAsset(const AssetLocation& location,
269 const ImporterInputVariant& inputVariant)
270 {
271 auto importers = getImportersForType<AssetType>();
272 if (importers.empty()) {
273 LOG(NEXO_ERROR, "No importers registered for asset type {}", typeid(AssetType).name());
274 return GenericAssetRef::null().as<AssetType>();
275 }
276 return importAssetTryImporters(location, inputVariant, importers).template as<AssetType>();
277 }
