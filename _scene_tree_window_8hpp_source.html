155 template <typename... Components, typename NodeCreator>
156 static void generateNodes(std::map<scene::SceneId, SceneObject>& scenes, NodeCreator nodeCreator)
157 {
158 const std::vector<ecs::Entity> entities = Application::m_coordinator->getAllEntitiesWith<Components...>();
159 for (const ecs::Entity entity : entities)
160 {
161 const auto& sceneTag = Application::m_coordinator->getComponent<components::SceneTag>(entity);
162 if (auto it = scenes.find(sceneTag.id); it != scenes.end())
163 {
164 SceneObject newNode = nodeCreator(it->second.data.sceneProperties.sceneId,
165 it->second.data.sceneProperties.windowId, entity);
166 it->second.children.push_back(newNode);
167 }
168 }
169 }
