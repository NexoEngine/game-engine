160 template <typename... Components, typename NodeCreator>
161 static void generateNodes(std::map<scene::SceneId, SceneObject>& scenes, NodeCreator nodeCreator)
162 {
163 const std::vector<ecs::Entity> entities = Application::m_coordinator->getAllEntitiesWith<Components...>();
164 for (const ecs::Entity entity : entities)
165 {
166 const auto& sceneTag = Application::m_coordinator->getComponent<components::SceneTag>(entity);
167 if (auto it = scenes.find(sceneTag.id); it != scenes.end())
168 {
169 SceneObject newNode = nodeCreator(it->second.data.sceneProperties.sceneId,
170 it->second.data.sceneProperties.windowId, entity);
171 it->second.children.push_back(newNode);
172 }
173 }
174 }
