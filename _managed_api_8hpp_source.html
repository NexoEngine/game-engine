38 template<typename Sig>
39 struct ManagedApiFn;
40
41 template<typename Ret, typename... Args>
42 struct ManagedApiFn<Ret(Args...)> {
43 using Type = Ret (CORECLR_DELEGATE_CALLTYPE *)(Args...);
44
45 // Constructor explicitly accepting function pointers
46 explicit(false) ManagedApiFn(const Type f) : func(f)
47 {
48 if (!func) {
49 THROW_EXCEPTION(InvalidManagedApi, std::format("Function pointer is null: {}", typeid(Type).name()));
50 }
51 }
52
53 Ret operator()(Args... args) const
54 {
55 assert(func != nullptr && "Called function pointer is null");
56 return func(args...);
57 }
58
59 explicit ManagedApiFn(nullptr_t) = delete;
60
61 // Delete the default constructor to enforce initialization
62 ManagedApiFn() = default;
63
64 private:
65 Type func = nullptr;
66 };
