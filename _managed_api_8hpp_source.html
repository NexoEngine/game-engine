37 template<typename Ret, typename... Args>
38 struct ManagedApiFn<Ret(Args...)> {
39 using Type = Ret (CORECLR_DELEGATE_CALLTYPE *)(Args...);
40
41 // Constructor explicitly accepting function pointers
42 explicit(false) ManagedApiFn(const Type f) : func(f)
43 {
44 if (!func) {
45 THROW_EXCEPTION(InvalidManagedApi, std::format("Function pointer is null: {}", typeid(Type).name()));
46 }
47 }
48
49 Ret operator()(Args... args) const
50 {
51 assert(func != nullptr && "Called function pointer is null");
52 return func(args...);
53 }
54
55 explicit ManagedApiFn(nullptr_t) = delete;
56
57 // Delete the default constructor to enforce initialization
58 ManagedApiFn() = default;
59
60 private:
61 Type func = nullptr;
62 };
