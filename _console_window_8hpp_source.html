164 template<typename... Args>
165 void addLog(std::format_string<Args...> fmt, Args&&... args)
166 {
167 try {
168 std::string formattedString = std::format(fmt, std::forward<Args>(args)...);
169
170 LogMessage newMessage;
171 newMessage.verbosity = loguru::Verbosity_1;
172 newMessage.message = formattedString;
173 newMessage.prefix = "";
174 m_logs.push_back(newMessage);
175 } catch (const std::exception &e) {
176 LogMessage newMessage;
177 newMessage.verbosity = loguru::Verbosity_ERROR;
178
179 char errorBuffer[1024];
180
181 // format up to sizeof(errorBuffer)-1 characters
182 auto result = std::format_to_n(
183 std::begin(errorBuffer),
184 std::size(errorBuffer) - 1,
185 "Error formatting log message: {}",
186 e.what()
187 );
188
189 // null-terminate
190 *result.out = '\0';
191
192 newMessage.message = std::string(errorBuffer);
193 newMessage.prefix = "";
194 m_logs.push_back(newMessage);
195 }
196
197 m_scrollToBottom = true;
198 }
