141 AssetLocation AssetImporterContext::genUniqueDependencyLocation()
142 {
143 auto depLoc = AssetLocation(location.getFullLocation());
144 depLoc.setName(formatUniqueName(location.getName().data(), AssetType::TYPE, ++m_depUniqueId));
145
146 if (!AssetCatalog::getInstance().getAsset(depLoc))
147 return depLoc;
148
149 // If the location already exists, we need to generate a new one
150 while (AssetCatalog::getInstance().getAsset(depLoc)) {
151 depLoc.setName(formatUniqueName(location.getName().data(), AssetType::TYPE, ++m_depUniqueId));
152 if (m_depUniqueId > ASSET_MAX_DEPENDENCIES) {
153 // Prevent infinite loop
154 LOG(NEXO_ERROR, "Failed to generate unique name for asset: {}: couldn't find unique id", depLoc.getFullLocation());
155 break;
156 }
157 }
158 return depLoc;
159 }
