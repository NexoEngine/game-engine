69 template<typename T>
70 struct has_save_method<T,
71 std::void_t<decltype(std::declval<const T&>().save())>>
72 : std::is_same<decltype(std::declval<const T&>().save()), typename T::Memento> {};

74 // Check if T::Memento has a restore() method that returns T

78 template<typename T>
79 struct has_restore_method<T,
80 std::void_t<decltype(std::declval<T&>().restore(std::declval<const typename T::Memento&>()))>>
81 : std::is_same<decltype(std::declval<T&>().restore(std::declval<const typename T::Memento&>())), void> {};

164 m_supportsMementoPattern.emplace(typeid(T), true);
165
166 m_saveComponentFunctions[typeid(T)] = [](const std::any& componentAny) -> std::any {
167 const T& component = std::any_cast<const T&>(componentAny);
168 return std::any(component.save());
169 };
170
171 m_restoreComponentFunctions[typeid(T)] = [](const std::any& mementoAny) -> std::any {
172 const typename T::Memento& memento = std::any_cast<const typename T::Memento&>(mementoAny);
173 T component{};
174 component.restore(memento);
175 return std::any(std::move(component));
176 };
177 } else {
178 m_supportsMementoPattern.emplace(typeid(T), false);

425 template<typename... Components>
426 std::vector<Entity> getAllEntitiesWith() const
427 {
428 // Prepare signatures
429 Signature requiredSignature;
430 Signature excludeSignature;
431
432 // Process each component type
433 (processComponentSignature<Components>(requiredSignature, excludeSignature), ...);
434
435 // Query entities
436 std::span<const Entity> livingEntities = m_entityManager->getLivingEntities();
437 std::vector<Entity> result;
438 result.reserve(livingEntities.size());
439
440 for (Entity entity : livingEntities)
441 {
442 const Signature entitySignature = m_entityManager->getSignature(entity);
443
444 // Entity must have all required components
445 const bool hasAllRequired = (entitySignature & requiredSignature) == requiredSignature;
446
447 // Entity must not have any excluded components
448 const bool hasAnyExcluded = (entitySignature & excludeSignature).any();
449
450 if (hasAllRequired && !hasAnyExcluded)
451 result.push_back(entity);
452 }
453
454 return result;
455 }

486 template <typename T, typename... Args>
487 std::shared_ptr<T> registerQuerySystem(Args&&... args) {
488 auto newQuerySystem = m_systemManager->registerQuerySystem<T>(std::forward<Args>(args)...);
489 std::span<const Entity> livingEntities = m_entityManager->getLivingEntities();
490 const Signature querySystemSignature = newQuerySystem->getSignature();
491 for (Entity entity : livingEntities) {
492 const Signature entitySignature = m_entityManager->getSignature(entity);
493 if ((entitySignature & querySystemSignature) == querySystemSignature) {
494 newQuerySystem->entities.insert(entity);
495 }
496 }
497 return newQuerySystem;

624 void processComponentSignature(Signature& required, Signature& excluded) const {
625 if constexpr (is_exclude_v<Component>) {
