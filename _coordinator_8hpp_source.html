155 };
156 m_typeIDtoTypeIndex.emplace(getComponentType<T>(), typeid(T));

158 m_addComponentFunctions[typeid(T)] = [this](Entity entity, const std::any& componentAny) {
159 T component = std::any_cast<T>(componentAny);
160 this->addComponent<T>(entity, component);
161 };
162
163 if constexpr (supports_memento_pattern_v<T>) {
164 m_supportsMementoPattern.emplace(typeid(T), true);
165
166 m_saveComponentFunctions[typeid(T)] = [](const std::any& componentAny) -> std::any {
167 const T& component = std::any_cast<const T&>(componentAny);
168 return std::any(component.save());
169 };
170
171 m_restoreComponentFunctions[typeid(T)] = [](const std::any& mementoAny) -> std::any {
172 const typename T::Memento& memento = std::any_cast<const typename T::Memento&>(mementoAny);
173 return std::any(memento.restore());
174 };
175 } else {
176 m_supportsMementoPattern.emplace(typeid(T), false);
177 }

423 template<typename... Components>
424 std::vector<Entity> getAllEntitiesWith() const
425 {
426 // Prepare signatures
427 Signature requiredSignature;
428 Signature excludeSignature;
429
430 // Process each component type
431 (processComponentSignature<Components>(requiredSignature, excludeSignature), ...);
432
433 // Query entities
434 std::span<const Entity> livingEntities = m_entityManager->getLivingEntities();
435 std::vector<Entity> result;
436 result.reserve(livingEntities.size());
437
438 for (Entity entity : livingEntities)
439 {
440 const Signature entitySignature = m_entityManager->getSignature(entity);
441
442 // Entity must have all required components
443 bool hasAllRequired = (entitySignature & requiredSignature) == requiredSignature;
444
445 // Entity must not have any excluded components
446 bool hasAnyExcluded = (entitySignature & excludeSignature).any();
447
448 if (hasAllRequired && !hasAnyExcluded)
449 result.push_back(entity);
450 }
451
452 return result;
453 }

484 template <typename T, typename... Args>
485 std::shared_ptr<T> registerQuerySystem(Args&&... args) {
486 auto newQuerySystem = m_systemManager->registerQuerySystem<T>(std::forward<Args>(args)...);
487 std::span<const Entity> livingEntities = m_entityManager->getLivingEntities();
488 const Signature querySystemSignature = newQuerySystem->getSignature();
489 for (Entity entity : livingEntities) {
490 const Signature entitySignature = m_entityManager->getSignature(entity);
491 if ((entitySignature & querySystemSignature) == querySystemSignature) {
492 newQuerySystem->entities.insert(entity);
493 }
494 }
495 return newQuerySystem;

622 void processComponentSignature(Signature& required, Signature& excluded) const {

624 // This is an excluded component
