208 template<typename... Components>
209 std::set<Entity> getAllEntitiesWith() const
210 {
211 Signature requiredSignature;
212 (requiredSignature.set(m_componentManager->getComponentType<Components>(), true), ...);
213
214 std::uint32_t checkedEntities = 0;
215 std::uint32_t livingEntities = m_entityManager->getLivingEntityCount();
216 std::set<Entity> result;
217 for (Entity i = 0; i < MAX_ENTITIES; ++i)
218 {
219 Signature entitySignature = m_entityManager->getSignature(i);
220 if (entitySignature.none())
221 continue;
222 if ((entitySignature & requiredSignature) == requiredSignature)
223 result.insert(i);
224 checkedEntities++;
225 if (checkedEntities > livingEntities)
226 break;
227 }
228 return result;
229 }
