126 void registerListener(L* listener) {
127 const std::type_index typeIndex(typeid(EventType));
128 if (!m_listeners.contains(typeIndex)) {
129 m_listeners[typeIndex] = std::vector<BaseListener*>();
130 }
131
132 m_listeners[typeIndex].push_back(listener);
133 LOG(NEXO_DEV, "EventManager(registerListeners): Registered listener {}", listener->getListenerName());
134 }

144 template <typename EventType>
145 void unregisterListener(BaseListener* listener) {
146 const std::type_index typeIndex(typeid(EventType));
147
148 if (const auto it = m_listeners.find(typeIndex); it != m_listeners.end()) {
149 auto& listeners = it->second;
150
151 if (const auto listenerIt = std::ranges::find(listeners, listener); listenerIt != listeners.end()) {
152 listeners.erase(listenerIt);
153 LOG(NEXO_DEV, "EventManager(unregisterListener): Unregistered listener {}", listener->getListenerName());
154
155 if (listeners.empty()) {
156 m_listeners.erase(it);
157 }
158 } else {
159 LOG(NEXO_WARN, "EventManager(unregisterListener): Listener not found for this event type");
160 }
161 } else {
162 LOG(NEXO_WARN, "EventManager(unregisterListener): No listeners found for this event type");
163 }
164 }
