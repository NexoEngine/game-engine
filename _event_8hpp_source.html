130 void registerListener(L* listener) {
131 const std::type_index typeIndex(typeid(EventType));
132 if (!m_listeners.contains(typeIndex)) {
133 m_listeners[typeIndex] = std::vector<BaseListener*>();
134 }
135
136 m_listeners[typeIndex].push_back(listener);
137 LOG(NEXO_DEV, "EventManager(registerListeners): Registered listener {}", listener->getListenerName());
138 }
139
148 template <typename EventType>
149 void unregisterListener(BaseListener* listener) {
150 const std::type_index typeIndex(typeid(EventType));
151
152 if (const auto it = m_listeners.find(typeIndex); it != m_listeners.end()) {
153 auto& listeners = it->second;
154
155 if (const auto listenerIt = std::ranges::find(listeners, listener); listenerIt != listeners.end()) {
156 listeners.erase(listenerIt);
157 LOG(NEXO_DEV, "EventManager(unregisterListener): Unregistered listener {}", listener->getListenerName());
158
159 if (listeners.empty()) {
160 m_listeners.erase(it);
161 }
162 } else {
163 LOG(NEXO_WARN, "EventManager(unregisterListener): Listener not found for this event type");
164 }
165 } else {
166 LOG(NEXO_WARN, "EventManager(unregisterListener): No listeners found for this event type");
167 }
168 }
