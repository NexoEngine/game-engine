116 template <typename EventType>
117 void registerListener(BaseListener* listener) {
118 const std::type_index typeIndex(typeid(EventType));
119 if (!m_listeners.contains(typeIndex)) {
120 m_listeners[typeIndex] = std::vector<BaseListener*>();
121 }
122
123 m_listeners[typeIndex].push_back(listener);
124 LOG(NEXO_DEV, "EventManager(registerListeners): Registered listener {}", listener->getListenerName());
125 }

135 template <typename EventType>
136 void unregisterListener(BaseListener* listener) {
137 const std::type_index typeIndex(typeid(EventType));
138
139 if (const auto it = m_listeners.find(typeIndex); it != m_listeners.end()) {
140 auto& listeners = it->second;
141
142 if (const auto listenerIt = std::ranges::find(listeners, listener); listenerIt != listeners.end()) {
143 listeners.erase(listenerIt);
144 LOG(NEXO_DEV, "EventManager(unregisterListener): Unregistered listener {}", listener->getListenerName());
145
146 if (listeners.empty()) {
147 m_listeners.erase(it);
148 }
149 } else {
150 LOG(NEXO_WARN, "EventManager(unregisterListener): Listener not found for this event type");
151 }
152 } else {
153 LOG(NEXO_WARN, "EventManager(unregisterListener): No listeners found for this event type");
154 }
155 }
