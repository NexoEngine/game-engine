65 inline ComponentType generateComponentTypeID()

67 assert(globalComponentCounter < MAX_COMPONENT_TYPE && "Maximum number of component types exceeded");
68 return globalComponentCounter++;
69 }

81 template<typename T>
82 ComponentType getUniqueComponentTypeID()
83 {
84 // This static variable is instantiated once per type T,
85 // but it will be assigned a unique value from the shared global counter.
86 // TODO: Warning! This is not thread-safe. (and it's a crappy implementation, but it works for now)
87 static const ComponentType id = generateComponentTypeID();
88 return id;
89 }

100 return getUniqueComponentTypeID<std::remove_cvref_t<T>>();
