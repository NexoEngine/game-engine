108 template <typename T, typename... Args>
109 void registerSingletonComponent(Args&&... args)
110 {
111 ComponentType typeName = getUniqueComponentTypeID<T>();
112 if (m_singletonComponents.contains(typeName)) {
113 LOG(NEXO_WARN, "ECS::SingletonComponentManager::registerSingletonComponent: trying to register a singleton component more than once");
114 return;
115 }
116 m_singletonComponents.insert(
117 {typeName,
118 std::make_shared<SingletonComponent<T>>(std::forward<Args>(args)...)}
119 );
120 }

129 template <typename T>
130 T &getSingletonComponent()
131 {
132 const ComponentType typeName = getUniqueComponentTypeID<T>();
133 if (!m_singletonComponents.contains(typeName))
134 THROW_EXCEPTION(SingletonComponentNotRegistered);
135
136 auto componentPtr = dynamic_cast<SingletonComponent<T>*>(m_singletonComponents[typeName].get());
137
138 return componentPtr->getInstance();
139 }

148 template <typename T>
149 std::shared_ptr<ISingletonComponent> getRawSingletonComponent()
150 {
151 const ComponentType typeName = getUniqueComponentTypeID<T>();
152 if (!m_singletonComponents.contains(typeName))
153 THROW_EXCEPTION(SingletonComponentNotRegistered);
154
155 return m_singletonComponents[typeName];
156 }
157
166 template <typename T>
167 void unregisterSingletonComponent()
168 {
169 const ComponentType typeName = getUniqueComponentTypeID<T>();
170 if (!m_singletonComponents.contains(typeName))
171 THROW_EXCEPTION(SingletonComponentNotRegistered);
172
173 m_singletonComponents.erase(typeName);
174 }
