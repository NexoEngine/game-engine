105 void registerSingletonComponent(Args&&... args)
106 {
107 ComponentType typeName = getUniqueComponentTypeID<T>();
108 if (m_singletonComponents.contains(typeName)) {
109 LOG(NEXO_WARN, "ECS::SingletonComponentManager::registerSingletonComponent: trying to register a singleton component more than once");
110 return;
111 }
112 m_singletonComponents.insert(

114 std::make_shared<SingletonComponent<T>>(std::forward<Args>(args)...)}

125 template <typename T>
126 T &getSingletonComponent()
127 {
128 const ComponentType typeName = getUniqueComponentTypeID<T>();
129 if (!m_singletonComponents.contains(typeName))
130 THROW_EXCEPTION(SingletonComponentNotRegistered);
131
132 auto componentPtr = dynamic_cast<SingletonComponent<T>*>(m_singletonComponents[typeName].get());
133
134 return componentPtr->getInstance();
135 }

164 {

166 if (!m_singletonComponents.contains(typeName))
