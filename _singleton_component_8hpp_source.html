92 using Decayed = std::decay_t<T>;
93 std::type_index typeName(typeid(Decayed));
94 if (m_singletonComponents.contains(typeName)) {
95 LOG(NEXO_WARN, "ECS::SingletonComponentManager::registerSingletonComponent: trying to register a singleton component more than once");
96 return;
97 }
98 m_singletonComponents.insert({typeName, std::make_shared<SingletonComponent<Decayed>>(std::forward<Args>(args)...)});
99 }

108 template <typename T>
109 T &getSingletonComponent() {
110 const std::type_index typeName(typeid(T));
111 if (!m_singletonComponents.contains(typeName))
112 THROW_EXCEPTION(SingletonComponentNotRegistered);
113
114 auto componentPtr = dynamic_cast<SingletonComponent<T>*>(m_singletonComponents[typeName].get());
115
116 return componentPtr->getInstance();
117 }

127 template <typename T>
128 void unregisterSingletonComponent() {
129 const std::type_index typeName(typeid(T));
130 if (!m_singletonComponents.contains(typeName))
131 THROW_EXCEPTION(SingletonComponentNotRegistered);
132 m_singletonComponents.erase(typeName);
133 }
134 private:
