64 template<typename T>
65 class ComponentArray final : public IComponentArray {
66 public:
73 void insertData(const Entity entity, T component)
74 {
75 if (m_size == MAX_ENTITIES)
76 THROW_EXCEPTION(OutOfRange, MAX_ENTITIES);
77 if (m_entityToIndexMap.contains(entity))
78 {
79 LOG(NEXO_WARN, "ECS::ComponentArray::insertData: Component already added to entity {}", entity);
80 return;
81 }
82
83 size_t newIndex = m_size;
84 m_entityToIndexMap[entity] = newIndex;
85 m_indexToEntityMap[newIndex] = entity;
86 m_componentArray[newIndex] = component;
87 ++m_size;
88 }
89
95 void removeData(const Entity entity)
96 {
97 if (!m_entityToIndexMap.contains(entity))
98 THROW_EXCEPTION(ComponentNotFound, entity);
99
100 size_t indexOfRemovedEntity = m_entityToIndexMap[entity];
101 size_t indexOfLastElement = m_size - 1;
102 m_componentArray[indexOfRemovedEntity] = m_componentArray[indexOfLastElement];
103
104 const Entity entityOfLastElement = m_indexToEntityMap[indexOfLastElement];
105 m_entityToIndexMap[entityOfLastElement] = indexOfRemovedEntity;
106 m_indexToEntityMap[indexOfRemovedEntity] = entityOfLastElement;
107
108 m_entityToIndexMap.erase(entity);
109 m_indexToEntityMap.erase(indexOfLastElement);
110
111 --m_size;
112 }
113
120 T &getData(const Entity entity)
121 {
122 if (!m_entityToIndexMap.contains(entity))
123 THROW_EXCEPTION(ComponentNotFound, entity);
124
125 return m_componentArray[m_entityToIndexMap[entity]];
126 }
127
133 void entityDestroyed(const Entity entity) override
134 {
135 if (m_entityToIndexMap.contains(entity))
136 removeData(entity);
137 }
138
139 bool hasComponent(const Entity entity) const
140 {
141 return m_entityToIndexMap.contains(entity);
142 }
143
144 private:
145 std::array<T, MAX_ENTITIES> m_componentArray;
146
147 std::unordered_map<Entity, size_t> m_entityToIndexMap;
148
149 std::unordered_map<size_t, Entity> m_indexToEntityMap;
150
151 size_t m_size = 0;
152 };

171 template<typename T>
172 void registerComponent()
173 {
174 std::type_index typeName(typeid(T));
175 if (m_componentTypes.contains(typeName))
176 {
177 LOG(NEXO_WARN, "ECS::ComponentManager::registerComponent: Component already registered");
178 return;
179 }
180
181 m_componentTypes.insert({typeName, _nextComponentType});
182
183 m_componentArrays.insert({typeName, std::make_shared<ComponentArray<T> >()});
184
185 ++_nextComponentType;
186 }
