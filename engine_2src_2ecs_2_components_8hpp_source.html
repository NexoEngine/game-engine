124 std::stringstream ss;
125 ss << "Owned: {";
126 bool first = true;
127
128 // Add owned component IDs
129 for (ComponentType i = 0; i < MAX_COMPONENT_TYPE; ++i) {

131 if (!first)
132 ss << ", ";
133 ss << "Component#" << i;
134 first = false;
135 }
136 }
137
138 ss << "}, Non-owned: {";
139 first = true;
140
141 // Add non-owned component IDs
142 for (ComponentType i = 0; i < MAX_COMPONENT_TYPE; ++i) {
143 if (nonOwnedSignature.test(i)) {
144 if (!first)
145 ss << ", ";
146 ss << "Component#" << i;
147 first = false;
148 }
149 }
150
151 ss << "}";
152 return ss.str();
153 }

556 template<typename... Owned>
557 auto registerGroup(const auto& nonOwned)
558 {
559 const GroupKey newGroupKey = generateGroupKey<Owned...>(nonOwned);
560
561 // Check if this exact group already exists
562 auto it = m_groupRegistry.find(newGroupKey);
563 if (it != m_groupRegistry.end()) {
564 using OwnedTuple = std::tuple<std::shared_ptr<ComponentArray<Owned>>...>;
565 using NonOwnedTuple = decltype(getNonOwnedTuple(nonOwned));
566 return std::static_pointer_cast<Group<OwnedTuple, NonOwnedTuple>>(it->second);
567 }
568
569 // Check for conflicts with existing groups
570 for (const auto& existingKey : std::ranges::views::keys(m_groupRegistry)) {
571 if (hasCommonOwnedComponents(existingKey, newGroupKey)) {
572 for (ComponentType i = 0; i < MAX_COMPONENT_TYPE; i++) {
573 if (existingKey.ownedSignature.test(i) && newGroupKey.ownedSignature.test(i)) {
574 THROW_EXCEPTION(OverlappingGroupsException,
575 existingKey.toString(),
576 newGroupKey.toString(),
577 i);
578 }
579 }
580 }
581 }
582
583 auto group = createNewGroup<Owned...>(nonOwned);
584 m_groupRegistry[newGroupKey] = group;
585 return group;
586 }

663 template<typename... Owned>
664 auto createNewGroup(const auto& nonOwned)
665 {
666 auto nonOwnedArrays = getNonOwnedTuple(nonOwned);
667
668 auto ownedArrays = std::make_tuple(getComponentArray<Owned>()...);
669 using OwnedTuple = std::tuple<std::shared_ptr<ComponentArray<Owned>>...>;
670 using NonOwnedTuple = decltype(nonOwnedArrays);
671
672 // Find entities that should be in this group
673 auto driver = std::get<0>(ownedArrays);
674 const std::size_t minSize = std::apply([](auto&&... arrays) -> std::size_t {
675 return std::min({ static_cast<std::size_t>(arrays->size())... });
676 }, ownedArrays);
677
678 for (std::size_t i = 0; i < minSize; ++i) {
679 Entity e = driver->getEntityAtIndex(i);
680 bool valid = true;
681
682 // Check in owned arrays
683 std::apply([&](auto&&... arrays) {
684 valid = (valid && ... && arrays->hasComponent(e));
685 }, ownedArrays);
686
687 // Check in non-owned arrays
688 std::apply([&](auto&&... arrays) {
689 valid = (valid && ... && arrays->hasComponent(e));
690 }, nonOwnedArrays);
691
692 if (valid) {
693 std::apply([&](auto&&... arrays) {
694 ((arrays->addToGroup(e)), ...);
695 }, ownedArrays);
696 }
697 }
698
699 return std::make_shared<Group<OwnedTuple, NonOwnedTuple>>(ownedArrays, nonOwnedArrays);
700 }
