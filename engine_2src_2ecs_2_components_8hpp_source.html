120 std::stringstream ss;

134 ss << "}, Non-owned: {";
135 first = true;
136
137 // Add non-owned component IDs
138 for (ComponentType i = 0; i < MAX_COMPONENT_TYPE; ++i) {

144 }
145 }
146
147 ss << "}";

552 template<typename... Owned>
553 auto registerGroup(const auto& nonOwned)
554 {
555 const GroupKey newGroupKey = generateGroupKey<Owned...>(nonOwned);
556
557 // Check if this exact group already exists
558 auto it = m_groupRegistry.find(newGroupKey);
559 if (it != m_groupRegistry.end()) {
560 using OwnedTuple = std::tuple<std::shared_ptr<ComponentArray<Owned>>...>;
561 using NonOwnedTuple = decltype(getNonOwnedTuple(nonOwned));
562 return std::static_pointer_cast<Group<OwnedTuple, NonOwnedTuple>>(it->second);
563 }
564
565 // Check for conflicts with existing groups
566 for (const auto& existingKey : std::ranges::views::keys(m_groupRegistry)) {
567 if (hasCommonOwnedComponents(existingKey, newGroupKey)) {
568 for (ComponentType i = 0; i < MAX_COMPONENT_TYPE; i++) {
569 if (existingKey.ownedSignature.test(i) && newGroupKey.ownedSignature.test(i)) {
570 THROW_EXCEPTION(OverlappingGroupsException,
571 existingKey.toString(),
572 newGroupKey.toString(),
573 i);
574 }
575 }
576 }
577 }
578
579 auto group = createNewGroup<Owned...>(nonOwned);
580 m_groupRegistry[newGroupKey] = group;
581 return group;
582 }

659 template<typename... Owned>
660 auto createNewGroup(const auto& nonOwned)
661 {
662 auto nonOwnedArrays = getNonOwnedTuple(nonOwned);
663
664 auto ownedArrays = std::make_tuple(getComponentArray<Owned>()...);
665 using OwnedTuple = std::tuple<std::shared_ptr<ComponentArray<Owned>>...>;
666 using NonOwnedTuple = decltype(nonOwnedArrays);
667
668 // Find entities that should be in this group
669 auto driver = std::get<0>(ownedArrays);
670 const std::size_t minSize = std::apply([](auto&&... arrays) -> std::size_t {
671 return std::min({ static_cast<std::size_t>(arrays->size())... });
672 }, ownedArrays);
673
674 for (std::size_t i = 0; i < minSize; ++i) {
675 Entity e = driver->getEntityAtIndex(i);
676 bool valid = true;
677
678 // Check in owned arrays
679 std::apply([&](auto&&... arrays) {
680 valid = (valid && ... && arrays->hasComponent(e));
681 }, ownedArrays);
682
683 // Check in non-owned arrays
684 std::apply([&](auto&&... arrays) {
685 valid = (valid && ... && arrays->hasComponent(e));
686 }, nonOwnedArrays);
687
688 if (valid) {
689 std::apply([&](auto&&... arrays) {
690 ((arrays->addToGroup(e)), ...);
691 }, ownedArrays);
692 }
693 }
694
695 return std::make_shared<Group<OwnedTuple, NonOwnedTuple>>(ownedArrays, nonOwnedArrays);
696 }
