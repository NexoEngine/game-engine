112 template<typename GetNameFunc>
113 bool RowEntityDropdown(
114 const std::string& label,
115 nexo::ecs::Entity& targetEntity,
116 const std::vector<nexo::ecs::Entity>& entities,
117 GetNameFunc&& getNameFunc
118 )
119 {
120 ImGui::TableNextRow();
121 ImGui::TableNextColumn();
122 ImGui::AlignTextToFramePadding();
123 ImGui::TextUnformatted(label.c_str());
124
125 ImGui::TableNextColumn();
126 IdGuard idGuard(label);
127
128 bool changed = false;
129
130 // Build entity-name mapping
131 static std::vector<std::pair<nexo::ecs::Entity, std::string>> entityNamePairs;
132 static nexo::ecs::Entity lastTargetEntity = 0;
133 static std::vector<nexo::ecs::Entity> lastEntities;
134
135 // Only rebuild the mapping if entities list changed or target entity changed
136 bool needRebuild = lastTargetEntity != targetEntity || lastEntities.size() != entities.size();
137
138 if (!needRebuild) {
139 for (size_t i = 0; i < entities.size() && !needRebuild; i++) {
140 needRebuild = lastEntities[i] != entities[i];
141 }
142 }
143
144 if (needRebuild) {
145 entityNamePairs.clear();
146 entityNamePairs.reserve(entities.size());
147 lastEntities = entities;
148 lastTargetEntity = targetEntity;
149
150 for (nexo::ecs::Entity entity : entities) {
151 std::string name = getNameFunc(entity);
152 entityNamePairs.emplace_back(entity, name);
153 }
154 }
155
156 // Find current index
157 int currentIndex = -1;
158 for (size_t i = 0; i < entityNamePairs.size(); i++) {
159 if (entityNamePairs[i].first == targetEntity) {
160 currentIndex = static_cast<int>(i);
161 break;
162 }
163 }
164
165 // Add a "None" option if we want to allow null selection
166 const std::string currentItemName = currentIndex >= 0 ? entityNamePairs[currentIndex].second : "None";
167
168 // Draw the combo box
169 ImGui::SetNextItemWidth(-FLT_MIN); // Use all available width
170 if (ImGui::BeginCombo("##entity_dropdown", currentItemName.c_str()))
171 {
172 // Optional: Add a "None" option for clearing the target
173 if (ImGui::Selectable("None", targetEntity == nexo::ecs::MAX_ENTITIES)) {
174 targetEntity = nexo::ecs::MAX_ENTITIES;
175 changed = true;
176 }
177
178 for (size_t i = 0; i < entityNamePairs.size(); i++)
179 {
180 const bool isSelected = (currentIndex == static_cast<int>(i));
181 if (ImGui::Selectable(entityNamePairs[i].second.c_str(), isSelected))
182 {
183 targetEntity = entityNamePairs[i].first;
184 changed = true;
185 }
186
187 if (isSelected)
188 ImGui::SetItemDefaultFocus();
189 }
190 ImGui::EndCombo();
191 }
192 if (ImGui::IsItemActive())
193 setItemActive();
194 if (ImGui::IsItemActivated())
195 setItemActivated();
196 if (ImGui::IsItemDeactivated())
197 setItemDeactivated();
198
199 return changed;
200 }
