149 template<typename T, size_t capacity = 1024>
150 requires (capacity >= 1)
151 class alignas(64) ComponentArray final : public IComponentArray {
152 public:
156 using component_type = T;
157
164 ComponentArray()
165 {
166 m_sparse.resize(capacity, INVALID_ENTITY);
167 m_dense.reserve(capacity);
168 m_componentArray.reserve(capacity);
169 }

209 void insert(Entity entity, T component)
210 {
211 if (entity >= MAX_ENTITIES)
212 THROW_EXCEPTION(OutOfRange, entity);
213
214 // Ensure m_sparse can hold this entity index.
215 ensureSparseCapacity(entity);
216
217 if (hasComponent(entity)) {
218 LOG(NEXO_WARN, "Entity {} already has component: {}", entity, typeid(T).name());
219 return;
220 }
221
222 const size_t newIndex = m_size;
223 m_sparse[entity] = newIndex;
224 m_dense.push_back(entity);
225 m_componentArray.push_back(component);
226
227 ++m_size;
228 }
229
240 void insertRaw(Entity entity, const void *componentData) override
241 {
242 if (entity >= MAX_ENTITIES)
243 THROW_EXCEPTION(OutOfRange, entity);
244
245 ensureSparseCapacity(entity);
246
247 if (hasComponent(entity)) {
248 LOG(NEXO_WARN, "Entity {} already has component: {}", entity, typeid(T).name());
249 return;
250 }
251
252 const size_t newIndex = m_size;
253 m_sparse[entity] = newIndex;
254 m_dense.push_back(entity);
255
256 // allocate new component in the array
257 m_componentArray.emplace_back();
258 // copy the raw data into the new component, if it is trivially copyable, use memcpy, otherwise use placement new
259 if constexpr (std::is_trivially_copyable_v<T>) {
260 std::memcpy(&m_componentArray[newIndex], componentData, sizeof(T));
261 ++m_size;
262 } else {
263 THROW_EXCEPTION(InternalError, "Component type is not trivially copyable, raw insertion is not supported");
264 }
265 }
266
278 void remove(const Entity entity) override
279 {
280 if (!hasComponent(entity))
281 THROW_EXCEPTION(ComponentNotFound, entity);
282
283 size_t indexToRemove = m_sparse[entity];
284
285 // If the entity is part of the group, remove it from the group first.
286 if (indexToRemove < m_groupSize) {
287 // Swap with the last grouped element if not already at the end.
288 size_t groupLastIndex = m_groupSize - 1;
289 if (indexToRemove != groupLastIndex) {
290 std::swap(m_componentArray[indexToRemove], m_componentArray[groupLastIndex]);
291 std::swap(m_dense[indexToRemove], m_dense[groupLastIndex]);
292 m_sparse[m_dense[indexToRemove]] = indexToRemove;
293 m_sparse[m_dense[groupLastIndex]] = groupLastIndex;
294 }
295 --m_groupSize;
296 indexToRemove = groupLastIndex;
297 }
298
299 // Standard removal from the overall array:
300 const size_t lastIndex = m_size - 1;
301 if (indexToRemove != lastIndex) {
302 std::swap(m_componentArray[indexToRemove], m_componentArray[lastIndex]);
303 std::swap(m_dense[indexToRemove], m_dense[lastIndex]);
304 m_sparse[m_dense[indexToRemove]] = indexToRemove;
305 }
306 m_sparse[entity] = INVALID_ENTITY;
307 m_componentArray.pop_back();
308 m_dense.pop_back();
309 --m_size;
310
311 shrinkIfNeeded();
312 }

443 void addToGroup(const Entity entity)
444 {
445 if (!hasComponent(entity))
446 THROW_EXCEPTION(ComponentNotFound, entity);
447
448 size_t index = m_sparse[entity];
449 if (index < m_groupSize)
450 return;
451 // Swap with the element at the group boundary.
452 if (index != m_groupSize) {
453 std::swap(m_componentArray[index], m_componentArray[m_groupSize]);
454 std::swap(m_dense[index], m_dense[m_groupSize]);
455 m_sparse[m_dense[index]] = index;
456 m_sparse[m_dense[m_groupSize]] = m_groupSize;
457 }
458 ++m_groupSize;
459 }
460
472 void removeFromGroup(const Entity entity)
473 {
474 if (!hasComponent(entity))
475 THROW_EXCEPTION(ComponentNotFound, entity);
476
477 size_t index = m_sparse[entity];
478 if (index >= m_groupSize)
479 return;
480 --m_groupSize;
481 if (index != m_groupSize) {
482 std::swap(m_componentArray[index], m_componentArray[m_groupSize]);
483 std::swap(m_dense[index], m_dense[m_groupSize]);
484 m_sparse[m_dense[index]] = index;
485 m_sparse[m_dense[m_groupSize]] = m_groupSize;
486 }
487 }

499 void forceSetComponentAt(size_t index, const Entity entity, T component)
500 {

504 m_sparse[entity] = index;

521 template<typename EntityIt, typename CompIt>
522 void insertBatch(EntityIt entitiesBegin, EntityIt entitiesEnd, CompIt componentsBegin)
523 {
524 CompIt compIt = componentsBegin;
525 for (EntityIt entityIt = entitiesBegin; entityIt != entitiesEnd; ++entityIt, ++compIt) {
526 insert(*entityIt, *compIt);
527 }

580 private:
581 // Dense storage for components.
582 std::vector<T> m_componentArray;
583 // Sparse mapping: maps entity ID to index in the dense arrays.
584 std::vector<size_t> m_sparse;
585 // Dense storage for entity IDs.
586 std::vector<Entity> m_dense;
587 // Current number of active components.
588 size_t m_size = 0;
589 // The first m_groupSize entries in m_dense/m_componentArray are considered "grouped".
590 size_t m_groupSize = 0;
591
597 void ensureSparseCapacity(const Entity entity)
598 {
599 if (entity >= m_sparse.size()) {
600 size_t newSize = m_sparse.size();
601 if (newSize == 0)
602 newSize = capacity;
603 while (entity >= newSize)
604 newSize *= 2;
605 m_sparse.resize(newSize, INVALID_ENTITY);
606 }
607 }
608
615 void shrinkIfNeeded()
616 {
617 if (m_size < m_componentArray.capacity() / 4 && m_componentArray.capacity() > capacity * 2) {
618 // Only shrink if vectors are significantly oversized to avoid frequent reallocations
619 size_t newCapacity = std::max(m_size * 2, capacity);
620 if (newCapacity < capacity)
621 newCapacity = capacity;
622
623 m_componentArray.shrink_to_fit();
624 m_dense.shrink_to_fit();
625
626 // Reserve the optimized capacity to ensure future growth is efficient
627 m_componentArray.reserve(newCapacity);
628 m_dense.reserve(newCapacity);
629 }
630 }
