205 void insert(Entity entity, T component)
206 {
207 if (entity >= MAX_ENTITIES)
208 THROW_EXCEPTION(OutOfRange, entity);
209
210 // Ensure m_sparse can hold this entity index.
211 ensureSparseCapacity(entity);
212
213 if (hasComponent(entity)) {
214 LOG(NEXO_WARN, "Entity {} already has component: {}", entity, typeid(T).name());
215 return;
216 }
217
218 const size_t newIndex = m_size;
219 m_sparse[entity] = newIndex;
220 m_dense.push_back(entity);
221 m_componentArray.push_back(component);
222
223 ++m_size;
224 }

236 void insertRaw(Entity entity, const void *componentData) override
237 {
238 if (entity >= MAX_ENTITIES)
239 THROW_EXCEPTION(OutOfRange, entity);
240
241 ensureSparseCapacity(entity);
242
243 if (hasComponent(entity)) {
244 LOG(NEXO_WARN, "Entity {} already has component: {}", entity, typeid(T).name());
245 return;
246 }
247
248 const size_t newIndex = m_size;
249 m_sparse[entity] = newIndex;
250 m_dense.push_back(entity);
251
252 // allocate new component in the array
253 m_componentArray.emplace_back();
254 // copy the raw data into the new component, if it is trivially copyable, use memcpy, otherwise use placement new
255 if constexpr (std::is_trivially_copyable_v<T>) {
256 std::memcpy(&m_componentArray[newIndex], componentData, sizeof(T));
257 ++m_size;
258 } else {
259 THROW_EXCEPTION(InternalError, "Component type is not trivially copyable, raw insertion is not supported");
260 }
261 }

274 void remove(const Entity entity) override
275 {
276 if (!hasComponent(entity))
277 THROW_EXCEPTION(ComponentNotFound, entity);
278
279 size_t indexToRemove = m_sparse[entity];
280
281 // If the entity is part of the group, remove it from the group first.
282 if (indexToRemove < m_groupSize) {
283 // Swap with the last grouped element if not already at the end.
284 size_t groupLastIndex = m_groupSize - 1;
285 if (indexToRemove != groupLastIndex) {
286 std::swap(m_componentArray[indexToRemove], m_componentArray[groupLastIndex]);
287 std::swap(m_dense[indexToRemove], m_dense[groupLastIndex]);
288 m_sparse[m_dense[indexToRemove]] = indexToRemove;
289 m_sparse[m_dense[groupLastIndex]] = groupLastIndex;
290 }
291 --m_groupSize;
292 indexToRemove = groupLastIndex;
293 }
294
295 // Standard removal from the overall array:
296 const size_t lastIndex = m_size - 1;
297 if (indexToRemove != lastIndex) {
298 std::swap(m_componentArray[indexToRemove], m_componentArray[lastIndex]);
299 std::swap(m_dense[indexToRemove], m_dense[lastIndex]);
300 m_sparse[m_dense[indexToRemove]] = indexToRemove;
301 }
302 m_sparse[entity] = INVALID_ENTITY;
303 m_componentArray.pop_back();
304 m_dense.pop_back();
305 --m_size;
306
307 shrinkIfNeeded();
308 }

439 void addToGroup(const Entity entity)
440 {
441 if (!hasComponent(entity))
442 THROW_EXCEPTION(ComponentNotFound, entity);
443
444 size_t index = m_sparse[entity];
445 if (index < m_groupSize)
446 return;
447 // Swap with the element at the group boundary.
448 if (index != m_groupSize) {
449 std::swap(m_componentArray[index], m_componentArray[m_groupSize]);
450 std::swap(m_dense[index], m_dense[m_groupSize]);
451 m_sparse[m_dense[index]] = index;
452 m_sparse[m_dense[m_groupSize]] = m_groupSize;
453 }
454 ++m_groupSize;
455 }

593 void ensureSparseCapacity(const Entity entity)
594 {
595 if (entity >= m_sparse.size()) {
596 size_t newSize = m_sparse.size();
597 if (newSize == 0)
598 newSize = capacity;
599 while (entity >= newSize)
600 newSize *= 2;
601 m_sparse.resize(newSize, INVALID_ENTITY);
602 }
603 }
604
611 void shrinkIfNeeded()
612 {
613 if (m_size < m_componentArray.capacity() / 4 && m_componentArray.capacity() > capacity * 2) {
614 // Only shrink if vectors are significantly oversized to avoid frequent reallocations
615 size_t newCapacity = std::max(m_size * 2, capacity);
616 if (newCapacity < capacity)
617 newCapacity = capacity;
618
619 m_componentArray.shrink_to_fit();
620 m_dense.shrink_to_fit();
621
622 // Reserve the optimized capacity to ensure future growth is efficient
623 m_componentArray.reserve(newCapacity);
624 m_dense.reserve(newCapacity);
625 }
