200 void insert(Entity entity, T component)
201 {
202 if (entity >= MAX_ENTITIES)
203 THROW_EXCEPTION(OutOfRange, entity);
204
205 // Ensure m_sparse can hold this entity index.
206 ensureSparseCapacity(entity);
207
208 if (hasComponent(entity)) {
209 LOG(NEXO_WARN, "Entity {} already has component: {}", entity, typeid(T).name());
210 return;
211 }
212
213 const size_t newIndex = m_size;
214 m_sparse[entity] = newIndex;
215 m_dense.push_back(entity);
216 m_componentArray.push_back(component);
217
218 ++m_size;
219 }

231 void insertRaw(Entity entity, const void *componentData) override
232 {
233 if constexpr (!std::is_trivially_copyable_v<T>) {
234 THROW_EXCEPTION(InternalError, "Component type must be trivially copyable for raw insertion");
235 }
236 if (entity >= MAX_ENTITIES)
237 THROW_EXCEPTION(OutOfRange, entity);
238
239 ensureSparseCapacity(entity);
240
241 if (hasComponent(entity)) {
242 LOG(NEXO_WARN, "Entity {} already has component: {}", entity, typeid(T).name());
243 return;
244 }
245
246 const size_t newIndex = m_size;
247 m_sparse[entity] = newIndex;
248 m_dense.push_back(entity);
249
250 // allocate new component in the array
251 m_componentArray.emplace_back();
252 // copy the raw data into the new component
253 std::memcpy(&m_componentArray[newIndex], componentData, sizeof(T));
254
255 ++m_size;
256 }

269 void remove(const Entity entity) override
270 {
271 if (!hasComponent(entity))
272 THROW_EXCEPTION(ComponentNotFound, entity);
273
274 size_t indexToRemove = m_sparse[entity];
275
276 // If the entity is part of the group, remove it from the group first.
277 if (indexToRemove < m_groupSize) {
278 // Swap with the last grouped element if not already at the end.
279 size_t groupLastIndex = m_groupSize - 1;
280 if (indexToRemove != groupLastIndex) {
281 std::swap(m_componentArray[indexToRemove], m_componentArray[groupLastIndex]);
282 std::swap(m_dense[indexToRemove], m_dense[groupLastIndex]);
283 m_sparse[m_dense[indexToRemove]] = indexToRemove;
284 m_sparse[m_dense[groupLastIndex]] = groupLastIndex;
285 }
286 --m_groupSize;
287 indexToRemove = groupLastIndex;
288 }
289
290 // Standard removal from the overall array:
291 const size_t lastIndex = m_size - 1;
292 if (indexToRemove != lastIndex) {
293 std::swap(m_componentArray[indexToRemove], m_componentArray[lastIndex]);
294 std::swap(m_dense[indexToRemove], m_dense[lastIndex]);
295 m_sparse[m_dense[indexToRemove]] = indexToRemove;
296 }
297 m_sparse[entity] = INVALID_ENTITY;
298 m_componentArray.pop_back();
299 m_dense.pop_back();
300 --m_size;
301
302 shrinkIfNeeded();
303 }

434 void addToGroup(const Entity entity)
435 {
436 if (!hasComponent(entity))
437 THROW_EXCEPTION(ComponentNotFound, entity);
438
439 size_t index = m_sparse[entity];
440 if (index < m_groupSize)
441 return;
442 // Swap with the element at the group boundary.
443 if (index != m_groupSize) {
444 std::swap(m_componentArray[index], m_componentArray[m_groupSize]);
445 std::swap(m_dense[index], m_dense[m_groupSize]);
446 m_sparse[m_dense[index]] = index;
447 m_sparse[m_dense[m_groupSize]] = m_groupSize;
448 }
449 ++m_groupSize;
450 }

588 void ensureSparseCapacity(const Entity entity)
589 {
590 if (entity >= m_sparse.size()) {
591 size_t newSize = m_sparse.size();
592 if (newSize == 0)
593 newSize = capacity;
594 while (entity >= newSize)
595 newSize *= 2;
596 m_sparse.resize(newSize, INVALID_ENTITY);
597 }
598 }
599
606 void shrinkIfNeeded()
607 {
608 if (m_size < m_componentArray.capacity() / 4 && m_componentArray.capacity() > capacity * 2) {
609 // Only shrink if vectors are significantly oversized to avoid frequent reallocations
610 size_t newCapacity = std::max(m_size * 2, static_cast<size_t>(capacity));
611 if (newCapacity < capacity)
612 newCapacity = capacity;
613
614 m_componentArray.shrink_to_fit();
615 m_dense.shrink_to_fit();
616
617 // Reserve the optimized capacity to ensure future growth is efficient
618 m_componentArray.reserve(newCapacity);
619 m_dense.reserve(newCapacity);
620 }
