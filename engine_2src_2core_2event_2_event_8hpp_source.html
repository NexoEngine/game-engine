67 class EventManager {
68 public:
69 template <typename EventType>
70 void registerListener(BaseListener* listener) {
71 const std::type_index typeIndex(typeid(EventType));
72 if (!m_listeners.contains(typeIndex)) {
73 m_listeners[typeIndex] = std::vector<BaseListener*>();
74 }
75
76 m_listeners[typeIndex].push_back(listener);
77 LOG(NEXO_DEV, "EventManager(registerListeners): Registered listener {}", listener->getListenerName());
78 }
79
80 template <typename EventType>
81 void unregisterListener(BaseListener* listener) {
82 const std::type_index typeIndex(typeid(EventType));
83
84 if (const auto it = m_listeners.find(typeIndex); it != m_listeners.end()) {
85 auto& listeners = it->second;
86
87 if (const auto listenerIt = std::ranges::find(listeners, listener); listenerIt != listeners.end()) {
88 listeners.erase(listenerIt);
89 LOG(NEXO_DEV, "EventManager(unregisterListener): Unregistered listener {}", listener->getListenerName());
90
91 if (listeners.empty()) {
92 m_listeners.erase(it);
93 }
94 } else {
95 LOG(NEXO_WARN, "EventManager(unregisterListener): Listener not found for this event type");
96 }
97 } else {
98 LOG(NEXO_WARN, "EventManager(unregisterListener): No listeners found for this event type");
99 }
100 }
