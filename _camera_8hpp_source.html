64 [[nodiscard]] glm::mat4 getProjectionMatrix() const
65 {
66 if (type == CameraType::PERSPECTIVE) {
67 return glm::perspective(glm::radians(fov), static_cast<float>(width) / static_cast<float>(height), nearPlane, farPlane);
68 }
69 return glm::ortho(0.0f, static_cast<float>(width), static_cast<float>(height), 0.0f, nearPlane, farPlane);
70 }

81 [[nodiscard]] glm::mat4 getViewMatrix(const TransformComponent &transf) const
82 {
83 const glm::vec3 forward = transf.quat * glm::vec3(0.0f, 0.0f, -1.0f);
84 const glm::vec3 upVec = transf.quat * glm::vec3(0.0f, 1.0f, 0.0f);
85 return glm::lookAt(transf.pos, transf.pos + forward, upVec);
86 }
