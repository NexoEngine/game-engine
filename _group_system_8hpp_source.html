251 template<typename T>
252 auto get()
253 {
254 if (!m_group)
255 THROW_EXCEPTION(InternalError, "Group is null in GroupSystem");
256
257 constexpr bool isOwned = isOwnedComponent<T>();
258 if constexpr (isOwned) {
259 // Get the span from the group
260 auto baseSpan = m_group->template get<T>();
261
262 // Wrap it in our access-controlled span
263 return ComponentSpan<std::conditional_t<
264 GetComponentAccess<T>::accessType == AccessType::Read,
265 const T,
266 T
267 >>(baseSpan);
268 } else {
269 // For non-owned components, return the component array itself
270 auto componentArray = m_group->template get<T>();
271
272 // Apply access control by wrapping in a special pointer wrapper if needed
273 if constexpr (GetComponentAccess<T>::accessType == AccessType::Read)
274 return std::shared_ptr<const ComponentArray<T>>(m_group->template get<T>());
275 else
276 return componentArray;
277 }
278 }

323 template<typename... OT, typename... NOT>
324 std::shared_ptr<ActualGroupType> createGroupImpl(std::tuple<OT...>, std::tuple<NOT...>)
325 {
326 if constexpr (sizeof...(OT) > 0) {
327 if constexpr (sizeof...(NOT) > 0) {
328 auto group = coord->registerGroup<OT...>(nexo::ecs::get<NOT...>());
329 if (!group)
