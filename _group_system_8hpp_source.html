247 template<typename T>
248 auto get()
249 {
250 if (!m_group)
251 THROW_EXCEPTION(InternalError, "Group is null in GroupSystem");
252
253 constexpr bool isOwned = isOwnedComponent<T>();
254 if constexpr (isOwned) {
255 // Get the span from the group
256 auto baseSpan = m_group->template get<T>();
257
258 // Wrap it in our access-controlled span
259 return ComponentSpan<std::conditional_t<
260 GetComponentAccess<T>::accessType == AccessType::Read,
261 const T,
262 T
263 >>(baseSpan);
264 } else {
265 // For non-owned components, return the component array itself
266 auto componentArray = m_group->template get<T>();
267
268 // Apply access control by wrapping in a special pointer wrapper if needed
269 if constexpr (GetComponentAccess<T>::accessType == AccessType::Read)
270 return std::shared_ptr<const ComponentArray<T>>(m_group->template get<T>());
271 else
272 return componentArray;
273 }
274 }
