67 static unsigned int shaderDataTypeSize(NxShaderDataType type)
68 {
69 switch (type)
70 {
71 case NxShaderDataType::FLOAT: return 4; // 1 float (4 bytes)
72 case NxShaderDataType::FLOAT2: return 4 * 2; // 2 floats (8 bytes)
73 case NxShaderDataType::FLOAT3: return 4 * 3; // 3 floats (12 bytes)
74 case NxShaderDataType::FLOAT4: return 4 * 4; // 4 floats (16 bytes)
75 case NxShaderDataType::MAT3: return 4 * 3 * 3; // 3x3 matrix (36 bytes)
76 case NxShaderDataType::MAT4: return 4 * 4 * 4; // 4x4 matrix (64 bytes)
77 case NxShaderDataType::INT: return 4; // 1 int (4 bytes)
78 case NxShaderDataType::INT2: return 4 * 2; // 2 ints (8 bytes)
79 case NxShaderDataType::INT3: return 4 * 3; // 3 ints (12 bytes)
80 case NxShaderDataType::INT4: return 4 * 4; // 4 ints (16 bytes)
81 case NxShaderDataType::BOOL: return 1; // 1 byte (1 bool)
82 case NxShaderDataType::NONE: return 0; // No type, return 0
83 }
84 return 0; // Default case for undefined types
85 }

181 void calculateOffsetAndStride()
182 {
183 unsigned int offset = 0;
184 _stride = 0;
185 for (auto &element : _elements)
186 {
187 element.offset = offset;
188 offset += element.size;
189 _stride += element.size;
190 }
191 }
