71 static unsigned int shaderDataTypeSize(const NxShaderDataType type)
72 {
73 switch (type)
74 {
75 case NxShaderDataType::FLOAT: return 4; // 1 float (4 bytes)
76 case NxShaderDataType::FLOAT2: return 4 * 2; // 2 floats (8 bytes)
77 case NxShaderDataType::FLOAT3: return 4 * 3; // 3 floats (12 bytes)
78 case NxShaderDataType::FLOAT4: return 4 * 4; // 4 floats (16 bytes)
79 case NxShaderDataType::MAT3: return 4 * 3 * 3; // 3x3 matrix (36 bytes)
80 case NxShaderDataType::MAT4: return 4 * 4 * 4; // 4x4 matrix (64 bytes)
81 case NxShaderDataType::INT: return 4; // 1 int (4 bytes)
82 case NxShaderDataType::INT2: return 4 * 2; // 2 ints (8 bytes)
83 case NxShaderDataType::INT3: return 4 * 3; // 3 ints (12 bytes)
84 case NxShaderDataType::INT4: return 4 * 4; // 4 ints (16 bytes)
85 case NxShaderDataType::BOOL: return 1; // 1 byte (1 bool)
86 case NxShaderDataType::NONE: return 0; // No type, return 0
87 }
88 return 0; // Default case for undefined types
89 }

181 private:

184
185 void calculateOffsetAndStride()
186 {
187 unsigned int offset = 0;
188 _stride = 0;
189 for (auto &element : _elements)
190 {
191 element.offset = offset;
192 offset += element.size;
193 _stride += element.size;
194 }
195 }
