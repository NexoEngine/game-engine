26 template<typename T>
27 constexpr GLenum getGLTypeFromTemplate()
28 {
29 if constexpr (std::is_same_v<T, float>)
30 return GL_FLOAT;
31 else if constexpr (std::is_same_v<T, int>)
32 return GL_INT;
33 else if constexpr (std::is_same_v<T, unsigned int>)
34 return GL_UNSIGNED_INT;
35 else if constexpr (std::is_same_v<T, glm::vec4>)
36 return GL_RGBA;
37 return 0;
38 }

40 static int framebufferTextureFormatToOpenGlFormat(FrameBufferTextureFormats format)
41 {
42 constexpr GLenum formats[] = {GL_NONE, GL_RGBA, GL_RGBA, GL_RED_INTEGER};
43 if (static_cast<unsigned int>(format) == 0 || format >= FrameBufferTextureFormats::DEPTH24STENCIL8) // Maybe change that later
44 return -1;
45 return static_cast<int>(formats[static_cast<unsigned int>(format)]);
46 }

153 template<typename T>
154 T getPixelImpl(unsigned int attachmentIndex, int x, int y) const
155 {
156 if (attachmentIndex >= m_colorAttachments.size())
157 THROW_EXCEPTION(FramebufferInvalidIndex, "OPENGL", attachmentIndex);
158
159 glReadBuffer(GL_COLOR_ATTACHMENT0 + attachmentIndex);
160
161 auto &textureFormat = m_colorAttachmentsSpecs[attachmentIndex].textureFormat;
162 GLenum format = framebufferTextureFormatToOpenGlFormat(textureFormat);
163 GLenum type = getGLTypeFromTemplate<T>();
164
165 T pixelData;
166 glReadPixels(x, y, 1, 1, format, type, &pixelData);
167
168 return pixelData;
169 }
