26 template<typename T>
27 constexpr GLenum getGLTypeFromTemplate()
28 {
29 if constexpr (std::is_same_v<T, float>)
30 return GL_FLOAT;
31 else if constexpr (std::is_same_v<T, int>)
32 return GL_INT;
33 else if constexpr (std::is_same_v<T, unsigned int>)
34 return GL_UNSIGNED_INT;
35 else if constexpr (std::is_same_v<T, glm::vec4>)
36 return GL_RGBA;
37 return 0;
38 }

40 static int framebufferTextureFormatToOpenGlFormat(NxFrameBufferTextureFormats format)
41 {
42 constexpr GLenum formats[] = {GL_NONE, GL_RGBA, GL_RGBA, GL_RED_INTEGER};
43 if (static_cast<unsigned int>(format) == 0 || format >= NxFrameBufferTextureFormats::DEPTH24STENCIL8) // Maybe change that later
44 return -1;
45 return static_cast<int>(formats[static_cast<unsigned int>(format)]);
46 }

161 template<typename T>
162 T getPixelImpl(unsigned int attachmentIndex, int x, int y) const
163 {
164 if (attachmentIndex >= m_colorAttachments.size())
165 THROW_EXCEPTION(NxFramebufferInvalidIndex, "OPENGL", attachmentIndex);
166
167 glReadBuffer(GL_COLOR_ATTACHMENT0 + attachmentIndex);
168
169 auto &textureFormat = m_colorAttachmentsSpecs[attachmentIndex].textureFormat;
170 GLenum format = framebufferTextureFormatToOpenGlFormat(textureFormat);
171 GLenum type = getGLTypeFromTemplate<T>();
172
173 T pixelData;
174 glReadPixels(x, y, 1, 1, format, type, &pixelData);
175
176 return pixelData;
177 }

192 {
193 if (attachmentIndex >= m_colorAttachments.size())
194 THROW_EXCEPTION(NxFramebufferInvalidIndex, "OPENGL", attachmentIndex);
195 auto &spec = m_colorAttachmentsSpecs[attachmentIndex];
196 constexpr GLenum type = getGLTypeFromTemplate<T>();

198 glClearTexImage(m_colorAttachments[attachmentIndex], 0, framebufferTextureFormatToOpenGlFormat(spec.textureFormat), type, value);
