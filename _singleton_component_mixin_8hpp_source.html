95 template<typename T>
96 void cacheSingletonComponent()
97 {
98 auto* derived = static_cast<Derived*>(this);
99 const std::shared_ptr<ISingletonComponent> instance = derived->coord->template getRawSingletonComponent<T>();
100
101 // Store in the type-specific cache
102 auto typedInstance = std::static_pointer_cast<SingletonComponent<T>>(instance);

114 template<typename T>
115 static constexpr bool hasReadSingletonAccess()

117 return HasReadSingletonAccessImpl<T>::value;
118 }

128 template<typename T>
129 std::conditional_t<hasReadSingletonAccess<T>(), const T&, T&> getSingleton()
130 {
131 if (!TypedComponentCache<T>::instance) {
132 // Late binding in case the singleton was registered after system creation
133 cacheSingletonComponent<T>();
134 }
135
136 auto& typedComponent = TypedComponentCache<T>::instance;
137
138 if (!typedComponent)
139 THROW_EXCEPTION(SingletonComponentNotRegistered);
140
141 // Return the reference with appropriate constness
142 if constexpr (hasReadSingletonAccess<T>())
143 return const_cast<const T&>(typedComponent->getInstance());
144 else
145 return typedComponent->getInstance();
146 }
