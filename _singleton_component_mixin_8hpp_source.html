124 template<typename T>
125 std::conditional_t<hasReadSingletonAccess<T>(), const T&, T&> getSingleton()
126 {
127 if (!TypedComponentCache<T>::instance) {
128 // Late binding in case the singleton was registered after system creation
129 cacheSingletonComponent<T>();
130 }
131
132 auto& typedComponent = TypedComponentCache<T>::instance;
133
134 if (!typedComponent)
135 THROW_EXCEPTION(SingletonComponentNotRegistered);
136
137 // Return the reference with appropriate constness
138 if constexpr (hasReadSingletonAccess<T>())
139 return const_cast<const T&>(typedComponent->getInstance());
140 else
141 return typedComponent->getInstance();
142 }
