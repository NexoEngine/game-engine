151 void setLocation(const std::string& fullLocation)
152 {
153 std::string extractedPackName;
154 std::string extractedAssetName;
155 std::string extractedPath;
156
157 parseFullLocation(fullLocation, extractedAssetName, extractedPath, extractedPackName);
158 extractedPath = normalizePathAndRemovePrefixSlash(extractedPath);
159
160 try {
161 _name = AssetName(extractedAssetName);
162 if (!extractedPackName.empty())
163 _packName = AssetPackName(extractedPackName);
164 } catch (const InvalidName& e) {
165 THROW_EXCEPTION(InvalidAssetLocation, fullLocation, e.getMessage());
166 }

213 static void parseFullLocation(
214 std::string_view fullLocation,
215 std::string& extractedAssetName,
216 std::string& extractedPath,
217 std::string& extractedPackName
218 )
219 {
220 if (const auto packNameEndPos = fullLocation.find("::"); packNameEndPos != std::string::npos) {
221 extractedPackName = fullLocation.substr(0, packNameEndPos);
222 fullLocation.remove_prefix(packNameEndPos + 2);
223 } else {
224 extractedPackName.clear();
225 }
226 if (const auto pathStartPos = fullLocation.find('@'); pathStartPos != std::string::npos) {
227 extractedPath = fullLocation.substr(pathStartPos + 1);
228 fullLocation.remove_suffix(fullLocation.size() - pathStartPos);
229 } else {
230 extractedPath.clear();
231 }
232 // TODO: maybe trim spaces?
233 extractedAssetName = fullLocation;
234 }
