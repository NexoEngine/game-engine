157 void setLocation(const std::string& fullLocation)
158 {
159 std::string extractedPackName;
160 std::string extractedAssetName;
161 std::string extractedPath;
162
163 parseFullLocation(fullLocation, extractedAssetName, extractedPath, extractedPackName);
164 extractedPath = normalizePathAndRemovePrefixSlash(extractedPath);
165
166 try {
167 _name = AssetName(extractedAssetName);
168 if (!extractedPackName.empty())
169 _packName = AssetPackName(extractedPackName);
170 } catch (const InvalidName& e) {
171 THROW_EXCEPTION(InvalidAssetLocation, fullLocation, e.getMessage());
172 }
173 _path = extractedPath;
174 }

219 static void parseFullLocation(
220 std::string_view fullLocation,
221 std::string& extractedAssetName,
222 std::string& extractedPath,
223 std::string& extractedPackName
224 )
225 {
226 if (const auto packNameEndPos = fullLocation.find("::"); packNameEndPos != std::string::npos) {
227 extractedPackName = fullLocation.substr(0, packNameEndPos);
228 fullLocation.remove_prefix(packNameEndPos + 2);
229 } else {
230 extractedPackName.clear();
231 }
232 if (const auto pathStartPos = fullLocation.find('@'); pathStartPos != std::string::npos) {
233 extractedPath = fullLocation.substr(pathStartPos + 1);
234 fullLocation.remove_suffix(fullLocation.size() - pathStartPos);
235 } else {
236 extractedPath.clear();
237 }
238 // TODO: maybe trim spaces?
239 extractedAssetName = fullLocation;
240 }
