169 template<typename T, typename... Args>
170 std::shared_ptr<T> registerQuerySystem(Args&&... args)
171 {
172 static_assert(std::is_base_of_v<AQuerySystem, T>, "T must derive from AQuerySystem");
173 std::type_index typeName(typeid(T));
174
175 if (m_querySystems.contains(typeName))
176 {
177 LOG(NEXO_WARN, "ECS::SystemManager::registerSystem: trying to register a system more than once");
178 return nullptr;
179 }
180
181 auto system = std::make_shared<T>(std::forward<Args>(args)...);
182 m_querySystems.insert({typeName, system});
183 return system;
184 }
185
194 template<typename T, typename... Args>

198 std::type_index typeName(typeid(T));

202 LOG(NEXO_WARN, "ECS::SystemManager::registerSystem: trying to register a system more than once");
203 return nullptr;
204 }
205
206 auto system = std::make_shared<T>(std::forward<Args>(args)...);
207 m_groupSystems.insert({typeName, system});
