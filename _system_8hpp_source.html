173 template<typename T, typename... Args>
174 std::shared_ptr<T> registerQuerySystem(Args&&... args)
175 {
176 static_assert(std::is_base_of_v<AQuerySystem, T>, "T must derive from AQuerySystem");
177 std::type_index typeName(typeid(T));
178
179 if (m_querySystems.contains(typeName))
180 {
181 LOG(NEXO_WARN, "ECS::SystemManager::registerSystem: trying to register a system more than once");
182 return nullptr;
183 }
184
185 auto system = std::make_shared<T>(std::forward<Args>(args)...);
186 m_querySystems.insert({typeName, system});
187 return system;
188 }
189
