25 // Class to set up an OpenGL context
26 class OpenGLTest : public ::testing::Test {
27 protected:
28 GLFWwindow* window = nullptr;
29
30 void SetUp() override {
31 if (!glfwInit()) {
32 GTEST_FAIL() << "GLFW initialization failed. Failing OpenGL tests.";
33 }
34
35 glfwWindowHint(GLFW_CONTEXT_VERSION_MAJOR, 4);
36 glfwWindowHint(GLFW_CONTEXT_VERSION_MINOR, 5);
37 glfwWindowHint(GLFW_OPENGL_PROFILE, GLFW_OPENGL_CORE_PROFILE);
38
39 window = glfwCreateWindow(800, 600, "Test Window", nullptr, nullptr);
40 if (!window) {
41 glfwTerminate();
42 GTEST_FAIL() << "Failed to create GLFW window. Failing OpenGL tests.";
43 }
44
45 glfwMakeContextCurrent(window);
46
47 if (!gladLoadGLLoader((GLADloadproc)glfwGetProcAddress)) {
48 glfwDestroyWindow(window);
49 glfwTerminate();
50 GTEST_FAIL() << "Failed to initialize GLAD. Failing OpenGL tests.";
51 }
52
53 GLint major = 0, minor = 0;
54 glGetIntegerv(GL_MAJOR_VERSION, &major);
55 glGetIntegerv(GL_MINOR_VERSION, &minor);
56 if (major < 4 || (major == 4 && minor < 5)) {
57 glfwDestroyWindow(window);
58 glfwTerminate();
59 GTEST_FAIL() << "OpenGL 4.5 is required. Failing OpenGL tests.";
60 }
61 }
62
63 void TearDown() override {
64 if (window) {
65 glfwDestroyWindow(window);
66 }
67 glfwTerminate();
68 }
69 };
