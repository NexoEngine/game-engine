28 template<typename T>
29 auto toFormatFriendly(const T &value)
30 {
31 if constexpr (std::is_convertible_v<T, std::string_view>)
32 return std::string_view(value);
33 else
34 {
35 std::stringstream ss;
36 ss << value;
37 return ss.str();
38 }

51 inline std::string toString(const LogLevel level)

53 switch (level)

55 case LogLevel::FATAL: return "FATAL";

57 case LogLevel::WARN: return "WARN";
58 case LogLevel::INFO: return "INFO";

63 return "UNKNOWN";

66 inline std::string getFileName(const char *file)
67 {
68 std::string path(file);
69 if (const auto lastSlash = path.find_last_of("/\\"); lastSlash != std::string::npos)
70 return path.substr(lastSlash + 1);
71 return path;
72 }

74 inline void defaultCallback(const LogLevel level, const std::source_location& loc, const std::string &message)
75 {
76 std::ostream& outputStream = level == LogLevel::FATAL || level == LogLevel::ERR
77 ? std::cerr
78 : std::cout;
79
80 outputStream << "[" << toString(level) << "] " << getFileName(loc.file_name()) << ":" << loc.line() << " - " << message << std::endl;
81 }

145 template<typename... Args>
146 static void logWithFormat(const LogLevel level, const std::source_location loc, const std::string_view fmt, Args &&... args)
147 {
148 auto transformed = std::tuple{toFormatFriendly(std::forward<Args>(args))...};
149
150 const std::string message = std::apply(
151 [&](auto &&... transformedArgs) {
152 return std::vformat(fmt, std::make_format_args(transformedArgs...));
153 },
154 transformed);
155
156 logCallback(level, loc, message);
157 }

168 static std::string generateKey(const std::string_view fmt, const std::string& location, Args&&... args)
169 {

176 return ss.str();

192 template<typename... Args>
193 static void logOnce(const LogLevel level, const std::source_location loc,
194 const std::string_view fmt, const std::string& key, Args &&... args)
195 {
196 if (OnceRegistry::instance().shouldLog(key)) {
197 logWithFormat(level, loc, fmt, std::forward<Args>(args)...);
198 }
199 }

206 static void resetOnce(const std::string& key) {
207 OnceRegistry::instance().reset(key);

213 static void resetAllOnce() {
214 OnceRegistry::instance().resetAll();
215 }

217 private:
218 static inline std::function<void(LogLevel, const std::source_location& loc, const std::string &)> logCallback = defaultCallback;
