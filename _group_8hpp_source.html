493 template<typename CompType, typename FieldType>
494 void sortBy(FieldExtractor<CompType, FieldType> extractor, bool ascending = true)
495 {
496 SortingOrder sortingOrder = ascending ? SortingOrder::ASCENDING : SortingOrder::DESCENDING;
497
498 if (sortingOrder != m_sortingOrder) {
499 m_sortingOrder = sortingOrder;
500 m_sortingInvalidated = true;
501 }
502
503 if (!m_sortingInvalidated)
504 return;
505
506 std::shared_ptr<ComponentArray<CompType>> compArray;
507
508 if constexpr (tuple_contains_component_v<CompType, OwnedTuple>) {
509 compArray = getOwnedImpl<CompType>();
510 } else if constexpr (tuple_contains_component_v<CompType, NonOwnedTuple>) {
511 compArray = getNonOwnedImpl<CompType>();
512 } else {
513 static_assert(dependent_false<CompType>::value, "Component type not found in group");
514 }
515
516 if (!compArray)
517 THROW_EXCEPTION(InternalError, "Component array is null");
518
519 auto drivingArray = std::get<0>(m_ownedArrays);
520 const size_t groupSize = drivingArray->groupSize();
521
522 std::vector<Entity> entities;
523 entities.reserve(groupSize);
524
525 // Add all entities currently in the group
526 for (size_t i = 0; i < groupSize; i++)
527 entities.push_back(drivingArray->getEntityAtIndex(i));
528
529 // Sort entities based on the extracted field from the component
530 std::ranges::sort(entities,
531 [&](Entity a, Entity b) {
532 const auto& compA = compArray->get(a);
533 const auto& compB = compArray->get(b);
534 if (ascending)
535 return extractor(compA) < extractor(compB);
536 return extractor(compA) > extractor(compB);
537 });
538
539 reorderGroup(entities);
540 m_sortingInvalidated = false;
541 }

633 template<typename CompType, typename KeyType>
634 PartitionView<KeyType> getPartitionView(FieldExtractor<CompType, KeyType> keyExtractor)
635 {
636 std::string typeId = typeid(KeyType).name();
637 typeId += "_" + std::string(typeid(CompType).name());
638
639 EntityKeyExtractor<KeyType> entityKeyExtractor = [this, keyExtractor](Entity e) {
640 if constexpr (tuple_contains_component_v<CompType, OwnedTuple>) {
641 auto compArray = getOwnedImpl<CompType>();
642 if (!compArray)
643 THROW_EXCEPTION(InternalError, "Component array is null");
644
645 return keyExtractor(compArray->get(e));
646 } else if constexpr (tuple_contains_component_v<CompType, NonOwnedTuple>) {
647 auto compArray = getNonOwnedImpl<CompType>();
648 if (!compArray)
649 THROW_EXCEPTION(InternalError, "Component array is null");

651 return keyExtractor(compArray->get(e));
652 } else
653 static_assert(dependent_false<CompType>::value, "Component type not found in group");
654 };

656 return getEntityPartitionView<KeyType>(typeId, entityKeyExtractor);

667 template<typename KeyType>
668 PartitionView<KeyType> getEntityPartitionView(const std::string& partitionId,
669 EntityKeyExtractor<KeyType> keyExtractor)
670 {
671 // Check if we already have this partition view
672 auto it = m_partitionStorageMap.find(partitionId);
673 if (it == m_partitionStorageMap.end()) {
674 auto storage = std::make_unique<PartitionStorage<KeyType>>(this, keyExtractor);
675 auto* storagePtr = storage.get();
676 m_partitionStorageMap[partitionId] = std::move(storage);
677 storagePtr->rebuild();
678
679 return PartitionView<KeyType>(this, storagePtr->getPartitions());
680 }
681
682 // Get the existing storage and cast to the right type
683 auto* storage = static_cast<PartitionStorage<KeyType>*>(it->second.get());
684
685 if (storage->isDirty())
686 storage->rebuild();
687
688 return PartitionView<KeyType>(this, storage->getPartitions());
689 }

735 template<typename KeyType>
736 class PartitionStorage final : public IPartitionStorage {
737 public:
744 PartitionStorage(Group* group, EntityKeyExtractor<KeyType> keyExtractor)
745 : m_group(group), m_keyExtractor(std::move(keyExtractor)) {}
746
747 [[nodiscard]] bool isDirty() const override { return m_isDirty; }
748 void markDirty() override { m_isDirty = true; }
749
756 void rebuild() override
757 {
758 if (!m_isDirty)
759 return;
760 auto drivingArray = std::get<0>(m_group->m_ownedArrays);
761 const size_t groupSize = drivingArray->groupSize();
762
763 // Skip if no entities
764 if (groupSize == 0) {
765 m_partitions.clear();
766 m_isDirty = false;
767 return;
768 }
769
770 std::unordered_map<KeyType, std::vector<Entity>> keyToEntities;
771
772 for (size_t i = 0; i < groupSize; i++) {
773 Entity e = drivingArray->getEntityAtIndex(i);
774 KeyType key = m_keyExtractor(e);
775 keyToEntities[key].push_back(e);
776 }
777
778 m_partitions.clear();
779 m_partitions.reserve(keyToEntities.size());
780
781 std::vector<Entity> newOrder;
782 newOrder.reserve(groupSize);
783
784 size_t currentIndex = 0;
785 for (auto& [key, entities] : keyToEntities) {
786 Partition<KeyType> partition;
787 partition.key = key;
788 partition.startIndex = currentIndex;
789 partition.count = entities.size();
790 m_partitions.push_back(partition);
791
792 // Add these entities to the new order
793 newOrder.insert(newOrder.end(), entities.begin(), entities.end());
794
795 currentIndex += entities.size();
796 }
797
798 m_group->reorderGroup(newOrder);
799 m_isDirty = false;
800 }

838 template<typename ArrayPtr>
839 void reorderArray(ArrayPtr array, const std::vector<Entity>& newOrder) const
840 {
841 size_t groupSize = array->groupSize();
842 if (newOrder.size() != groupSize)
843 THROW_EXCEPTION(InternalError, "New order size doesn't match group size");
844
845 if (groupSize == 0)
846 return;
847
848 // Create a temporary storage for components
849 using CompType = typename std::decay_t<decltype(*array)>::component_type;
850 std::vector<CompType> tempComponents;
851 tempComponents.reserve(groupSize);
852
853 for (Entity e : newOrder)
854 tempComponents.push_back(array->get(e)); //Maybe we should not push back, does it make a copy ?
855
856 for (size_t i = 0; i < groupSize; i++) {
857 Entity e = newOrder[i];
858 array->forceSetComponentAt(i, e, std::move(tempComponents[i]));
859 }
860 }

868 auto dereference(std::size_t index) const
869 {
870 Entity entity = std::get<0>(m_ownedArrays)->getEntityAtIndex(index);
871
872 // Use std::forward_as_tuple to preserve references.
873 auto ownedData = std::apply([entity](auto&&... arrays) {
874 return std::forward_as_tuple(arrays->get(entity)...);

876 // We still need the entity by value, so use std::make_tuple for that.
877 return std::tuple_cat(std::make_tuple(entity), ownedData);
