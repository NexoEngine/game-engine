491 template<typename CompType, typename FieldType>
492 void sortBy(FieldExtractor<CompType, FieldType> extractor, bool ascending = true)
493 {
494 SortingOrder sortingOrder = ascending ? SortingOrder::ASCENDING : SortingOrder::DESCENDING;
495
496 if (sortingOrder != m_sortingOrder) {
497 m_sortingOrder = sortingOrder;
498 m_sortingInvalidated = true;
499 }
500
501 if (!m_sortingInvalidated)
502 return;
503
504 std::shared_ptr<ComponentArray<CompType>> compArray;
505
506 if constexpr (tuple_contains_component_v<CompType, OwnedTuple>) {
507 compArray = getOwnedImpl<CompType>();
508 } else if constexpr (tuple_contains_component_v<CompType, NonOwnedTuple>) {
509 compArray = getNonOwnedImpl<CompType>();
510 } else {
511 static_assert(dependent_false<CompType>::value, "Component type not found in group");
512 }
513
514 if (!compArray)
515 THROW_EXCEPTION(InternalError, "Component array is null");
516
517 auto drivingArray = std::get<0>(m_ownedArrays);
518 const size_t groupSize = drivingArray->groupSize();
519
520 std::vector<Entity> entities;
521 entities.reserve(groupSize);
522
523 // Add all entities currently in the group
524 for (size_t i = 0; i < groupSize; i++)
525 entities.push_back(drivingArray->getEntityAtIndex(i));
526
527 // Sort entities based on the extracted field from the component
528 std::ranges::sort(entities,
529 [&](Entity a, Entity b) {
530 const auto& compA = compArray->get(a);
531 const auto& compB = compArray->get(b);
532 if (ascending)
533 return extractor(compA) < extractor(compB);
534 return extractor(compA) > extractor(compB);
535 });
536
537 reorderGroup(entities);
538 m_sortingInvalidated = false;
539 }

631 template<typename CompType, typename KeyType>
632 PartitionView<KeyType> getPartitionView(FieldExtractor<CompType, KeyType> keyExtractor)
633 {
634 std::string typeId = typeid(KeyType).name();
635 typeId += "_" + std::string(typeid(CompType).name());
636
637 EntityKeyExtractor<KeyType> entityKeyExtractor = [this, keyExtractor](Entity e) {
638 if constexpr (tuple_contains_component_v<CompType, OwnedTuple>) {
639 auto compArray = getOwnedImpl<CompType>();
640 if (!compArray)
641 THROW_EXCEPTION(InternalError, "Component array is null");
642
643 return keyExtractor(compArray->get(e));
644 } else if constexpr (tuple_contains_component_v<CompType, NonOwnedTuple>) {
645 auto compArray = getNonOwnedImpl<CompType>();
646 if (!compArray)
647 THROW_EXCEPTION(InternalError, "Component array is null");
648
649 return keyExtractor(compArray->get(e));
650 } else
651 static_assert(dependent_false<CompType>::value, "Component type not found in group");
652 };
653
654 return getEntityPartitionView<KeyType>(typeId, entityKeyExtractor);
655 }

665 template<typename KeyType>
666 PartitionView<KeyType> getEntityPartitionView(const std::string& partitionId,
667 EntityKeyExtractor<KeyType> keyExtractor)
668 {
669 // Check if we already have this partition view
670 auto it = m_partitionStorageMap.find(partitionId);
671 if (it == m_partitionStorageMap.end()) {
672 auto storage = std::make_unique<PartitionStorage<KeyType>>(this, keyExtractor);
673 auto* storagePtr = storage.get();
674 m_partitionStorageMap[partitionId] = std::move(storage);
675 storagePtr->rebuild();
676
677 return PartitionView<KeyType>(this, storagePtr->getPartitions());
678 }
679
680 // Get the existing storage and cast to the right type
681 auto* storage = static_cast<PartitionStorage<KeyType>*>(it->second.get());
682
683 if (storage->isDirty())
684 storage->rebuild();
685
686 return PartitionView<KeyType>(this, storage->getPartitions());
687 }

733 template<typename KeyType>
734 class PartitionStorage final : public IPartitionStorage {
735 public:
742 PartitionStorage(Group* group, EntityKeyExtractor<KeyType> keyExtractor)
743 : m_group(group), m_keyExtractor(std::move(keyExtractor)) {}
744
745 [[nodiscard]] bool isDirty() const override { return m_isDirty; }
746 void markDirty() override { m_isDirty = true; }
747
754 void rebuild() override
755 {
756 if (!m_isDirty)
757 return;
758 auto drivingArray = std::get<0>(m_group->m_ownedArrays);
759 const size_t groupSize = drivingArray->groupSize();
760
761 // Skip if no entities
762 if (groupSize == 0) {
763 m_partitions.clear();
764 m_isDirty = false;
765 return;
766 }
767
768 std::unordered_map<KeyType, std::vector<Entity>> keyToEntities;
769
770 for (size_t i = 0; i < groupSize; i++) {
771 Entity e = drivingArray->getEntityAtIndex(i);
772 KeyType key = m_keyExtractor(e);
773 keyToEntities[key].push_back(e);
774 }
775
776 m_partitions.clear();
777 m_partitions.reserve(keyToEntities.size());
778
779 std::vector<Entity> newOrder;
780 newOrder.reserve(groupSize);
781
782 size_t currentIndex = 0;
783 for (auto& [key, entities] : keyToEntities) {
784 Partition<KeyType> partition;
785 partition.key = key;
786 partition.startIndex = currentIndex;
787 partition.count = entities.size();
788 m_partitions.push_back(partition);
789
790 // Add these entities to the new order
791 newOrder.insert(newOrder.end(), entities.begin(), entities.end());
792
793 currentIndex += entities.size();
794 }
795
796 m_group->reorderGroup(newOrder);
797 m_isDirty = false;
798 }

836 template<typename ArrayPtr>
837 void reorderArray(ArrayPtr array, const std::vector<Entity>& newOrder) const
838 {
839 size_t groupSize = array->groupSize();
840 if (newOrder.size() != groupSize)
841 THROW_EXCEPTION(InternalError, "New order size doesn't match group size");
842
843 if (groupSize == 0)
844 return;
845
846 // Create a temporary storage for components
847 using CompType = typename std::decay_t<decltype(*array)>::component_type;
848 std::vector<CompType> tempComponents;
849 tempComponents.reserve(groupSize);
850
851 for (Entity e : newOrder)
852 tempComponents.push_back(array->get(e)); //Maybe we should not push back, does it make a copy ?
853
854 for (size_t i = 0; i < groupSize; i++) {
855 Entity e = newOrder[i];
856 array->forceSetComponentAt(i, e, std::move(tempComponents[i]));
857 }
858 }

866 auto dereference(std::size_t index) const
867 {
868 Entity entity = std::get<0>(m_ownedArrays)->getEntityAtIndex(index);
869
870 // Use std::forward_as_tuple to preserve references.
871 auto ownedData = std::apply([entity](auto&&... arrays) {
872 return std::forward_as_tuple(arrays->get(entity)...);
873 }, m_ownedArrays);
874 // We still need the entity by value, so use std::make_tuple for that.
875 return std::tuple_cat(std::make_tuple(entity), ownedData);
